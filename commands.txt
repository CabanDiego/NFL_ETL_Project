This file is filled with commands which can be used to run this project.
Not all commands may be needed, but are here for reminders.

---Connect to EC2---
ssh -i {pem file name}.pem ubuntu@{publicIPV4} 

---Update Packages---
sudo apt update
sudo apt upgrade -y 

---Install Postgres
sudo apt install postgresql postgresql-contrib -y 

--Switch to Postgres Linux User
sudo -i -u postgres

---Run Postgres---
sudo systemctl start postgresql
sudo systemctl enable postgresql to run postgres

---Verify Postgress is Running---
sudo systemctl status postgresql 

---Restart Postgres---
sudo systemctl restart postgresql

---Quit Postgres---
\q or ctrl + d

---Exit EC2 SSH Session---
exit

-----POSTGRESSQL COMMANDS-----
---Create DB---
CREATE DATABASE {Name of DB};
---Create User---
CREATE USER {Username} WITH PASSWORD '{Password}';
---Grant Privileges---
GRANT ALL PRIVILEGES ON DATABASE {Name of DB} TO {Username};
